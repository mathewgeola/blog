    新增10G的swap

    dd if=/dev/zero of=swapfile bs=1M count=10240

# mkswap创建交换文件:
mkswap swapfile

# swapon激活

# 手动挂载
# vmhgfs-fuse .host:/ /mnt/hgfs -o allow_other

################################################################################
.host:/ /mnt/hgfs fuse.vmhgfs-fuse defaults,allow_other 0 0
################################################################################
